  QT      += widgets designer

  QTDIR_build {
  # This is only for the Qt build. Do not use externally. We mean it.
  PLUGIN_TYPE = designer
  PLUGIN_CLASS_NAME = TicTacToePlugin
  load(qt_plugin)
  CONFIG += install_ok
  } else {
  # Public example:

  TEMPLATE = lib
  CONFIG  += plugin

  TARGET = $$qtLibraryTarget($$TARGET)

  target.path = $$[QT_INSTALL_PLUGINS]/designer
  INSTALLS += target

  }
